```typescript
export type InputSource = 'fit-report' | 'measured' | 'derived' | 'estimated';
export type HandlebarType = 'drop' | 'flat';

export interface FitInputs {
  fitReport?: { saddleHeight?: number; saddleForeAft?: number; handlebarHeight?: number; handlebarReach?: number; };
  physical?: { saddleHeight?: number; saddleForeAft?: number; handlebarHeight?: number; handlebarReach?: number; crankLength?: number; handlebarType?: HandlebarType; dropBarHoodHeightOffset?: number; dropBarHoodReachOffset?: number; };
  body?: { inseam?: number; torso?: number; arm?: number; };
  frame?: { stack?: number; reach?: number; seatTubeAngle?: number; };
}

export interface AxisOutput {
  ideal_mm: number;
  achievable_mm: number;
  letter_position: string | null;
  out_of_range: boolean;
  direction?: 'above' | 'below';
  source: InputSource;
  confidence: 'high' | 'medium' | 'low';
}

export interface FitOutputs {
  saddleHeight: AxisOutput | null;
  saddleForeAft: AxisOutput | null;
  handlebarHeight: AxisOutput | null;
  handlebarReach: AxisOutput | null;
  allAxesOutOfRange?: boolean;
}
```

    2. **`mmToLetter(mm: number, lookupTable: Record<string, number | null>): string | null`** -- Iterate all entries, skip null values, find closest by minimum absolute difference. Return null if no non-null entries exist.

    3. **`buildAxisOutput(idealMm: number, range: { minMm: number; maxMm: number }, lookupTable: Record<string, number | null>, source: InputSource): AxisOutput`** -- Shared helper. Clamp idealMm to [range.minMm, range.maxMm]. Set out_of_range and direction. Call mmToLetter. Set confidence to 'high' if letter_position is non-null, 'low' if null.

    4. **`buildAxisOutputNullableRange(idealMm: number, range: { minMm: number | null; maxMm: number | null }, lookupTable: Record<string, number | null>, source: InputSource): AxisOutput`** -- Variant for handlebar reach where range bounds may be null. When min/max are null, skip clamping (achievable_mm = ideal_mm, out_of_range = false).

       - Use `buildAxisOutput` with `SADDLE_HEIGHT_RANGE` and `SADDLE_HEIGHT_LETTER_TO_MM`
       - Return null if no applicable input (per D-13)

    6. **`calculateSaddleForeAft(inputs: FitInputs): AxisOutput | null`**

       - Use `buildAxisOutput` with `SADDLE_FORE_AFT_RANGE` and `SADDLE_FORE_AFT_LETTER_TO_MM`
       - Note: seat tube angle calculation for deriving from frame geometry is a stretch goal. For now, direct measurement and fit report paths are sufficient. Frame-derived path can return null (frame geometry alone does not specify saddle setback without additional cockpit data).
       - Return null if no applicable input

    7. **`calculateHandlebarHeight(inputs: FitInputs): AxisOutput | null`**
       - Priority: fitReport.handlebarHeight (no offset -- fit report is final) > physical.handlebarHeight

On the Zwift Ride flat bar, the equivalent position is HIGHER by the offset amount: `target = handlebarHeight + (dropBarHoodHeightOffset ?? DEFAULT_DROP_BAR_HOOD_HEIGHT_OFFSET_MM)`
       - When physical and handlebarType === 'flat' or undefined: direct transfer, `target = handlebarHeight`
       - Use `buildAxisOutput` with `HANDLEBAR_HEIGHT_RANGE` and `HANDLEBAR_HEIGHT_LETTER_TO_MM`
       - Return null if no applicable input

    8. **`calculateHandlebarReach(inputs: FitInputs): AxisOutput | null`**
       - Priority: fitReport.handlebarReach (no offset) > physical.handlebarReach

       - When physical and handlebarType === 'flat' or undefined: direct transfer
       - Use `buildAxisOutputNullableRange` with `HANDLEBAR_REACH_RANGE` and `HANDLEBAR_REACH_LETTER_TO_MM`
       - Return null if no applicable input
