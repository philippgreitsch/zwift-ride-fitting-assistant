1. Create `src/store/fitStore.ts`:
   ```typescript
   import { create } from 'zustand'
   import type { FitInputs } from '../types/fit'

   type SkillLevel = 'beginner' | 'pro'

   interface FitStore {
     inputs: FitInputs
     skillLevel: SkillLevel | null
     currentStep: number  // 0 = skill selector, 1-4 = wizard steps, 5 = results
     setSkillLevel: (level: SkillLevel) => void
     updatePhysical: (patch: Partial<NonNullable<FitInputs['physical']>>) => void
     updateBody: (patch: Partial<NonNullable<FitInputs['body']>>) => void
     updateFrame: (patch: Partial<NonNullable<FitInputs['frame']>>) => void
     updateFitReport: (patch: Partial<NonNullable<FitInputs['fitReport']>>) => void
     setCurrentStep: (step: number) => void
   }

   export const useFitStore = create<FitStore>()((set) => ({
     inputs: {},
     skillLevel: null,
     currentStep: 0,
     setSkillLevel: (level) => set({ skillLevel: level }),
     updatePhysical: (patch) =>
       set((s) => ({ inputs: { ...s.inputs, physical: { ...s.inputs.physical, ...patch } } })),
     updateBody: (patch) =>
       set((s) => ({ inputs: { ...s.inputs, body: { ...s.inputs.body, ...patch } } })),
     updateFrame: (patch) =>
       set((s) => ({ inputs: { ...s.inputs, frame: { ...s.inputs.frame, ...patch } } })),
     updateFitReport: (patch) =>
       set((s) => ({ inputs: { ...s.inputs, fitReport: { ...s.inputs.fitReport, ...patch } } })),
     setCurrentStep: (step) => set({ currentStep: step }),
   }))

   export type { SkillLevel }
   ```

2. Create `src/components/wizard/fields/UnitInput.tsx`:
   - A wrapper around shadcn `<Input>` that renders a unit suffix ("mm" or degrees symbol) to the right of the input value.
   - Props: `unit: 'mm' | 'degrees'`, plus all standard shadcn Input props (id, placeholder, className, etc.).
   - Implementation: Wrap `<Input>` in a relative div. Place an absolute-positioned `<span>` with the unit text at `right-12px`, vertically centered, in `text-sm text-zinc-500 font-mono pointer-events-none`.
   - The Input gets `pr-12` right padding to prevent text from overlapping the unit suffix.
   - Input type is `"text"` with `inputMode="decimal"` to get numeric keyboard on mobile without browser-specific spinner buttons.
   - Do NOT use `type="number"` — it causes issues with empty string handling and shows browser spinners.

   - Field spacing: 16px gap between fields (Tailwind `space-y-4` on parent or `mb-4` on each field).
